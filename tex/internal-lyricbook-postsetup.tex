%% internal-lyricbook-presetup.tex
%% ===============================
%%
%% Part of the system for automatically creating lyrics-only version of a book.
%%
%% This file contains instructions for after loading the main .sty file,
%% and is automatically appended before \input{tex/setup_<...>.tex}
%% instruction in the main document.
%%
%% That \input instruction must exist in the main document, for being able to
%% automatically create lyrics-only books, and within the included setup_<...>
%% file must the main package unilaiva-songbook_common.sty be loaded.
%%
%% See internal-lyricbook-presetup.tex file also.
%%

\lyriconlytrue

\chordsoff
\measureson

\showkeyfalse
\showgoodkeysfalse
\shownotesfalse
\showbeatsfalse
\showlilypondfalse
\showaltchordsfalse

% Make the measure bar lighter
\definecolor{mbarcolor}{rgb}{.68,.44,.82}

% Make numbers in the meter smaller and the same color as mbar:
\renewcommand{\meterfont}{\color{mbarcolor}\tiny\sffamily\upshape}

\baselineadj=+1pt plus 0pt minus 0pt
\hypersetup{pdftitle={\thisbooktitle { (\variantlyricsonlytext)}}}
